def fifaclassified(TotalStats):
    if TotalStats > 85:
        classification = "Top Performer"
    else:
        classification = "Non Top-Performer"
    return classification

fifa23["player_classification"] = fifa23["TotalStats"].apply(fifaclassified)
print(fifa23[["Full Name", "TotalStats", "player_classification"]].head(10))
